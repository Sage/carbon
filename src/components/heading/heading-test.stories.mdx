import { Meta, Story, Canvas } from "@storybook/addon-docs";

import specialCharacters from "../../../.storybook/utils/argTypes/specialCharacters";
import Heading from "./heading.component";

<Meta
  title="Heading/Test"
  parameters={{
    info: { disable: true },
    chromatic: {
      disable: true,
    },
  }}
  argTypes={{
    titleSpecialCharacters: specialCharacters,
    childrenSpecialCharacters: specialCharacters,
    subheaderSpecialCharacters: specialCharacters,
    helpSpecialCharacters: specialCharacters,
    helpLinkSpecialCharacters: specialCharacters,
    backLinkSpecialCharacters: specialCharacters,
  }}
/>

export const HeadingStory = ({
  titleSpecialCharacters,
  title,
  childrenSpecialCharacters,
  children,
  subheaderSpecialCharacters,
  subheader,
  helpSpecialCharacters,
  help,
  helpLinkSpecialCharacters,
  helpLink,
  backLinkSpecialCharacters,
  backLink,
  ...args
}) => {
  return (
    <Heading
      title={title || titleSpecialCharacters}
      children={children || childrenSpecialCharacters}
      subheader={subheader || subheaderSpecialCharacters}
      help={help || helpSpecialCharacters}
      helpLink={helpLink || helpLinkSpecialCharacters}
      backLink={backLink || backLinkSpecialCharacters}
      {...args}
    />
  );
};
