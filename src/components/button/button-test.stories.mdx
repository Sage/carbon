export const commonArgTypes = {
  size: {

export const commonArgs = {
  size: "medium",
  children: "Example Button",
  disabled: false,
  fullWidth: false,
  subtext: "",
  buttonType: "secondary",
  href: undefined,
  destructive: false,
  noWrap: false,
  iconPosition: "before",
};

export const ButtonStory = ({ subtext, children, ...args }) => (
  <Button
    onClick={action("click")}
    subtext={subtext}
    children={children}
    {...args}
  />
);

export const ButtonAsASiblingStory = ({ subtext, children, ...args }) => {
  return (
    <Box>
      <Button
        subtext={subtext}
        children={children}
        {...args}
        onClick={action("click")}
      />
      <Button
        subtext={subtext}
        children={children}
        {...args}
        onClick={action("click")}
        ml={2}
      />
    </Box>
  );
};

export const generateButtons = (buttonType, iconPosition) => {
  return (
    <>
      {["", "bin"].map((iconType) => {
        const props = { iconPosition, buttonType, iconType };
        return (
          <Box key={`${buttonType}-${iconPosition}-${iconType}`}>
            {BUTTON_SIZES.map((size) => (
              <React.Fragment
                key={`${buttonType}-${iconPosition}-${iconType}-${size}`}
              >
                <Button key="basic" size={size} {...props} ml={2}>
                  {size}
                </Button>
                {size === "large" && (
                  <Button
                    key="subtext"
                    size={size}
                    subtext="line two"
                    {...props}
                    ml={2}
                  >
                    {size}
                  </Button>
                )}
              </React.Fragment>
            ))}
            {BUTTON_SIZES.map((size) => (
              <React.Fragment
                key={`${buttonType}-${iconPosition}-${iconType}-${size}-destructive`}
              >
                <Button key="basic" size={size} destructive {...props} ml={2}>
                  {size}
                </Button>
                {size === "large" && (
                  <Button
                    key="subtext"
                    size={size}
                    destructive
                    subtext="line two"
                    {...props}
                    ml={2}
                  >
                    {size}
                  </Button>
                )}
              </React.Fragment>
            ))}
            {BUTTON_SIZES.map((size) => (
              <React.Fragment
                key={`${buttonType}-${iconPosition}-${iconType}-${size}-disabled`}
              >
                <Button key="basic" size={size} disabled {...props} ml={2}>
                  {size}
                </Button>
                {size === "large" && (
                  <Button
                    key="subtext"
                    size={size}
                    disabled
                    subtext="line two"
                    {...props}
                    ml={2}
                  >
                    {size}
                  </Button>
                )}
              </React.Fragment>
            ))}
          </Box>
        );
      })}
    </>
  );
};

export const NoWrapButtonsStory = ({ noWrap }) => {
  return (
    <>
      {BUTTON_VARIANTS.map((buttonType) => {
        return BUTTON_SIZES.map((size) => {
          return (
            <Box width="100px">
              <Button buttonType={buttonType} noWrap={noWrap} size={size}>
                Long button text
              </Button>
              <Button
                buttonType={buttonType}
                noWrap={noWrap}
                size={size}
                iconType="bin"
              >
                Long button text
              </Button>
              <Button
                buttonType={buttonType}
                noWrap={noWrap}
                size={size}
                iconType="bin"
                iconPosition="after"
              >
                Long button text
              </Button>
              <Button
                buttonType={buttonType}
                noWrap={noWrap}
                size="large"
                iconType="bin"
                subtext="Even longer button subtext"
              >
                Long button text
              </Button>
              <Button
                buttonType={buttonType}
                noWrap={noWrap}
                size="large"
                iconType="bin"
                iconPosition="after"
                subtext="Even longer button subtext"
              >
                Long button text
              </Button>
            </Box>
          );
        });
      })}
    </>
  );
};
